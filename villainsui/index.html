-- Utility Functions
local function Create(instanceType, properties)
    local instance = Instance.new(instanceType)
    for prop, value in pairs(properties) do
        if prop ~= "Parent" then
            instance[prop] = value
        end
    end
    if properties.Parent then
        instance.Parent = properties.Parent
    end
    return instance
end

local function Tween(object, properties, duration, style, direction)
    local tweenInfo = TweenInfo.new(
        duration or 0.3,
        style or Enum.EasingStyle.Quint,
        direction or Enum.EasingDirection.Out
    )
    local tween = TweenService:Create(object, tweenInfo, properties)
    tween:Play()
    return tween
end

local function AddCorner(parent, radius)
    return Create("UICorner", {
        CornerRadius = radius or Theme.CornerRadius,
        Parent = parent
    })
end

local function AddStroke(parent, color, thickness)
    return Create("UIStroke", {
        Color = color or Theme.Primary,
        Thickness = thickness or 1,
        Transparency = 0.7,
        Parent = parent
    })
end

-- Key System Module
local KeySystem = {}
KeySystem.__index = KeySystem

function KeySystem.new(settings)
    local self = setmetatable({}, KeySystem)
    self.Keys = settings.Keys or {}
    self.KeyLink = settings.KeyLink or ""
    self.LinkType = settings.LinkType or "platoboost" -- platoboost, workink, linkvertise
    self.SaveKey = settings.SaveKey ~= false
    self.KeyName = settings.KeyName or "VillainsUI_Key"
    return self
end

function KeySystem:GetKeyLink()
    local links = {
        platoboost = "https://gateway.platoboost.com/a/YOUR_ID",
        workink = "https://work.ink/YOUR_LINK",
        linkvertise = "https://linkvertise.com/YOUR_ID"
    }
    return self.KeyLink ~= "" and self.KeyLink or links[self.LinkType]
end

function KeySystem:ValidateKey(inputKey)
    for _, key in ipairs(self.Keys) do
        if inputKey == key then
            if self.SaveKey then
                writefile(self.KeyName .. ".txt", inputKey)
            end
            return true
        end
    end
    return false
end

function KeySystem:CheckSavedKey()
    if self.SaveKey and isfile(self.KeyName .. ".txt") then
        local savedKey = readfile(self.KeyName .. ".txt")
        return self:ValidateKey(savedKey)
    end
    return false
end

function VillainsUI:CreateWindow(settings)
    local self = setmetatable({}, Window)
    
    self.Title = settings.Title or "Villains UI"
    self.Subtitle = settings.Subtitle or "by VillainsTeam"
    self.LogoId = settings.LogoId or "rbxassetid://0"
    self.KeySystem = settings.KeySystem or false
    self.Keys = settings.Keys or {}
    self.KeyLink = settings.KeyLink or ""
    self.LinkType = settings.LinkType or "platoboost"
    self.SaveKey = settings.SaveKey ~= false
    self.Size = settings.Size or UDim2.new(0, 600, 0, 450)
    self.Tabs = {}
    self.CurrentTab = nil
    self.Minimized = false
    self.FireEnabled = true
    
    -- Create ScreenGui
    local success, gui = pcall(function()
        local screenGui = Create("ScreenGui", {
            Name = "VillainsUI",
            ResetOnSpawn = false,
            ZIndexBehavior = Enum.ZIndexBehavior.Sibling
        })
        
        -- Try CoreGui first, fallback to PlayerGui
        pcall(function()
            screenGui.Parent = CoreGui
        end)
        
        if not screenGui.Parent then
            screenGui.Parent = Player:WaitForChild("PlayerGui")
        end
        
        return screenGui
    end)
    
    self.ScreenGui = gui
    
    -- Key System UI
    if self.KeySystem then
        self:CreateKeySystemUI()
    else
        self:CreateMainUI()
    end
    
    return self
end

function Window:StartFireAnimation()
    if not self.FireContainer then return end
    
    spawn(function()
        while self.FireEnabled and self.FireContainer and self.FireContainer.Parent do
            local particle = Create("Frame", {
                Size = UDim2.new(0, math.random(10, 25), 0, math.random(10, 25)),
                Position = UDim2.new(math.random(0, 100)/100, 0, 1, 0),
                BackgroundColor3 = Color3.fromRGB(
                    math.random(200, 255),
                    math.random(50, 100),
                    0
                ),
                BackgroundTransparency = 0.3,
                Parent = self.FireContainer
            })
            AddCorner(particle, UDim.new(1, 0))
            
            local duration = math.random(10, 20) / 10
            Tween(particle, {
                Position = UDim2.new(particle.Position.X.Scale, 0, -0.5, 0),
                Size = UDim2.new(0, 5, 0, 5),
                BackgroundTransparency = 1
            }, duration)
            
            game:GetService("Debris"):AddItem(particle, duration)
            wait(0.05)
        end
    end)
end

function Window:ToggleMinimize()
    self.Minimized = not self.Minimized
    
    if self.Minimized then
        Tween(self.MainFrame, {Size = UDim2.new(0, 0, 0, 0)}, 0.3)
        wait(0.3)
        self.MainFrame.Visible = false
        self.MinLogo.Visible = true
        Tween(self.MinLogo, {Size = UDim2.new(0, 50, 0, 50)}, 0.3)
    else
        self.MinLogo.Visible = false
        self.MainFrame.Visible = true
        Tween(self.MainFrame, {Size = self.Size}, 0.3)
    end
end

function Window:MakeDraggable(dragObject, targetObject)
    local dragging = false
    local dragInput, mousePos, framePos
    
    dragObject.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or 
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            mousePos = input.Position
            framePos = targetObject.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)
    
    dragObject.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or 
           input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - mousePos
            Tween(targetObject, {
                Position = UDim2.new(
                    framePos.X.Scale,
                    framePos.X.Offset + delta.X,
                    framePos.Y.Scale,
                    framePos.Y.Offset + delta.Y
                )
            }, 0.1)
        end
    end)
end

function Window:Destroy()

        wait(0.3)
        self.ScreenGui:Destroy()
    end
end

function Window:CreateTab(settings)
    local tab = setmetatable({}, Tab)
    
    tab.Name = settings.Name or "Tab"
    tab.Icon = settings.Icon or "rbxassetid://0"
    tab.Window = self
    tab.Elements = {}
    
    -- Tab Button
    local TabBtn = Create("TextButton", {
        Name = tab.Name,
        Size = UDim2.new(1, 0, 0, 40),
        BackgroundColor3 = Theme.Secondary,
        BackgroundTransparency = 0.5,
        Text = "",
        Parent = self.TabContainer
    })
    AddCorner(TabBtn, UDim.new(0, 8))
    
    local TabIcon = Create("ImageLabel", {
        Size = UDim2.new(0, 18, 0, 18),
        Position = UDim2.new(0, 10, 0.5, -9),
        BackgroundTransparency = 1,
        Image = tab.Icon,
        ImageColor3 = Theme.TextDark,
        Parent = TabBtn
    })
    
    local TabText = Create("TextLabel", {
        Size = UDim2.new(1, -40, 1, 0),
        Position = UDim2.new(0, 35, 0, 0),
        BackgroundTransparency = 1,
        Text = tab.Name,
        TextColor3 = Theme.TextDark,
        TextSize = 12,
        Font = Theme.Font,
        TextXAlignment = Enum.TextXAlignment.Left,
        Parent = TabBtn
    })
    
    tab.Button = TabBtn
    tab.Icon = TabIcon
    tab.Text = TabText
    
    -- Tab Content
    local TabContent = Create("ScrollingFrame", {
        Name = tab.Name .. "Content",
        Size = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        ScrollBarThickness = 3,
        ScrollBarImageColor3 = Theme.Primary,
        CanvasSize = UDim2.new(0, 0, 0, 0),
        AutomaticCanvasSize = Enum.AutomaticSize.Y,
        Visible = false,
        Parent = self.ContentContainer
    })
    
    local ContentLayout = Create("UIListLayout", {
        Padding = UDim.new(0, 8),
        Parent = TabContent
    })
    
    local ContentPadding = Create("UIPadding", {
        PaddingTop = UDim.new(0, 5),
        PaddingBottom = UDim.new(0, 5),
        PaddingLeft = UDim.new(0, 5),
        PaddingRight = UDim.new(0, 5),
        Parent = TabContent
    })
    
    tab.Content = TabContent
    
    -- Tab Selection
    TabBtn.MouseButton1Click:Connect(function()
        for _, t in ipairs(self.Tabs) do
            t.Content.Visible = false
            Tween(t.Button, {BackgroundTransparency = 0.5}, 0.2)
            Tween(t.Icon, {ImageColor3 = Theme.TextDark}, 0.2)
            t.Text.TextColor3 = Theme.TextDark
        end
        
        tab.Content.Visible = true
        Tween(TabBtn, {BackgroundTransparency = 0.3}, 0.2)
        Tween(TabIcon, {ImageColor3 = Theme.Primary}, 0.2)
        TabText.TextColor3 = Theme.Text
        
        self.CurrentTab = tab
    end)
    
    -- Hover Effects
    TabBtn.MouseEnter:Connect(function()
        if self.CurrentTab ~= tab then
            Tween(TabBtn, {BackgroundTransparency = 0.4}, 0.2)
        end
    end)
    
    TabBtn.MouseLeave:Connect(function()
        if self.CurrentTab ~= tab then
            Tween(TabBtn, {BackgroundTransparency = 0.5}, 0.2)
        end
    end)
    
    table.insert(self.Tabs, tab)
    
    -- Select first tab by default
    if #self.Tabs == 1 then
        TabBtn.BackgroundTransparency = 0.3
        TabIcon.ImageColor3 = Theme.Primary
        TabText.TextColor3 = Theme.Text
        TabContent.Visible = true
        self.CurrentTab = tab
    end
    
    return tab
end

    })

        Name = "Thumb",
        Size = UDim2.new(0, 18, 0, 18),
        Position = toggled and UDim2.new(1, -21, 0.5, -9) or UDim2.new(0, 3, 0.5, -9),
        BackgroundColor3 = Theme.Text,
        Parent = ToggleTrack
    })
    AddCorner(ToggleThumb, UDim.new(1, 0))
    
    local function UpdateToggle()
        toggled = not toggled
        
        Tween(ToggleThumb, {
            Position = toggled and UDim2.new(1, -21, 0.5, -9) or UDim2.new(0, 3, 0.5, -9)
        }, 0.3, Enum.EasingStyle.Back)
        
        Tween(ToggleTrack, {
            BackgroundColor3 = toggled and Theme.Primary or Theme.Secondary
        }, 0.2)
        
        if settings.Callback then
            settings.Callback(toggled)
        end
        
        self.Window:Notify(
            toggled and "Enabled" or "Disabled",
            settings.Name .. " is now " .. (toggled and "ON" or "OFF"),

        )

        BackgroundTransparency = 1,
        Text = "",
        Parent = Toggle
    })

    return {
        Set = function(value)
            if toggled ~= value then
                UpdateToggle()
            end
        end,
        Get = function()
            return toggled
        end
    }
end

-- Slider
function Tab:CreateSlider(settings)
    local value = settings.Default or settings.Min or 0
    local min = settings.Min or 0
    local max = settings.Max or 100
    local dragging = false
    
    local Slider = Create("Frame", {
        Name = "Slider_" .. settings.Name,
        Size = UDim2.new(1, 0, 0, 60),
        BackgroundColor3 = Theme.Secondary,
        Parent = self.Content
    })
    AddCorner(Slider)
    AddStroke(Slider, Theme.Primary, 1)
    
    local SliderText = Create("TextLabel", {
        Size = UDim2.new(1, -70, 0, 20),
        Position = UDim2.new(0, 15, 0, 8),
        BackgroundTransparency = 1,
        Text = settings.Name,
        TextColor3 = Theme.Text,
        TextSize = 13,
        Font = Theme.Font,
        TextXAlignment = Enum.TextXAlignment.Left,
        Parent = Slider
    })
    
    local SliderValue = Create("TextLabel", {
        Size = UDim2.new(0, 50, 0, 20),
        Position = UDim2.new(1, -60, 0, 8),
        BackgroundTransparency = 1,
        Text = tostring(value),
        TextColor3 = Theme.Primary,
        TextSize = 14,
        Font = Theme.Font,
        Parent = Slider
    })
    
    local SliderTrack = Create("Frame", {
        Name = "Track",
        Size = UDim2.new(1, -30, 0, 8),
        Position = UDim2.new(0, 15, 0, 40),
        BackgroundColor3 = Color3.fromRGB(40, 40, 40),
        Parent = Slider
    })
    AddCorner(SliderTrack, UDim.new(1, 0))
    
    local SliderFill = Create("Frame", {
        Name = "Fill",
        Size = UDim2.new((value - min) / (max - min), 0, 1, 0),
        BackgroundColor3 = Theme.Primary,
        Parent = SliderTrack
    })
    AddCorner(SliderFill, UDim.new(1, 0))
    
    local SliderThumb = Create("Frame", {
        Name = "Thumb",
        Size = UDim2.new(0, 16, 0, 16),
        Position = UDim2.new((value - min) / (max - min), -8, 0.5, -8),
        BackgroundColor3 = Theme.Text,
        ZIndex = 2,
        Parent = SliderTrack
    })
    AddCorner(SliderThumb, UDim.new(1, 0))
    
    local SliderBtn = Create("TextButton", {
        Size = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        Text = "",
        Parent = SliderTrack
    })
    
    local function UpdateSlider(input)
        local pos = math.clamp(
            (input.Position.X - SliderTrack.AbsolutePosition.X) / SliderTrack.AbsoluteSize.X,
            0, 1
        )
        value = math.floor(min + (max - min) * pos)
        
        Tween(SliderFill, {Size = UDim2.new(pos, 0, 1, 0)}, 0.1)
        Tween(SliderThumb, {Position = UDim2.new(pos, -8, 0.5, -8)}, 0.1)
        SliderValue.Text = tostring(value)
        
        if settings.Callback then
            settings.Callback(value)
        end
    end
    
    SliderBtn.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or 
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            UpdateSlider(input)
        end
    end)
    
    SliderBtn.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or 
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or 
           input.UserInputType == Enum.UserInputType.Touch) then
            UpdateSlider(input)
        end
    end)
    
    return {
        Set = function(newValue)
            value = math.clamp(newValue, min, max)
            local pos = (value - min) / (max - min)
            SliderFill.Size = UDim2.new(pos, 0, 1, 0)
            SliderThumb.Position = UDim2.new(pos, -8, 0.5, -8)
            SliderValue.Text = tostring(value)
        end,
        Get = function()
            return value
        end
    }
end

        local targetRotation = opened and 180 or 0
        
        Tween(Dropdown, {Size = targetSize}, 0.3)
        Tween(DropdownArrow, {Rotation = targetRotation}, 0.3)
    end)
    
    return {
        Set = function(option)
            if table.find(settings.Options, option) then
                selected = option
                DropdownSelected.Text = option
            end
        end,
        Get = function()
            return selected
        end,
        Refresh = function(newOptions)
            for _, child in ipairs(OptionsContainer:GetChildren()) do
                if child:IsA("TextButton") then
                    child:Destroy()
                end
            end
            settings.Options = newOptions
            -- Rebuild options...
        end
    }
end
